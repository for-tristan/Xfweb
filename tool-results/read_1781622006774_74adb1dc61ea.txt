    94→// Character-by-character reveal component (Vaulta SplitText style)
    95→// Reveals once on scroll into view — does NOT re-hide on scroll away to prevent lag spikes.
    96→//
    97→// Performance notes:
    98→//  - Default stagger reduced 0.02s -> 0.012s so the whole word finishes animating
    99→//    in <0.4s instead of running 1s+ (which made characters appear to "fly" while
   100→//    the user kept scrolling).
   101→//  - Per-char filter:blur is GPU-expensive; we keep it but cap the blur radius
   102→//    (see globals.css) and let CSS `contain` isolate the layout/paint work.
   103→function BlurText({ text, tag = 'span', className = '', stagger = 0.012 }: { text: string; tag?: string; className?: string; stagger?: number }) {
   104→  const ref = useRef<HTMLElement>(null);
   105→  const [revealed, setRevealed] = useState(false);
   106→
   107→  useEffect(() => {
   108→    const el = ref.current;
   109→    if (!el) return;
   110→    const observer = new IntersectionObserver(([entry]) => {
   111→      if (entry.isIntersecting) {
   112→        setRevealed(true);
   113→        observer.disconnect();
   114→      }
   115→    }, { threshold: 0.1 });
   116→    observer.observe(el);
   117→    return () => observer.disconnect();
   118→  }, []);
   119→
   120→  const chars = text.split('').map((char, i) => (
   121→    <span
   122→      key={i}
   123→      className={`blur-char${char === ' ' ? ' space' : ''}`}
   124→      style={revealed ? { transitionDelay: `${i * stagger}s` } : undefined}
   125→    >
   126→      {char === ' ' ? '\u00A0' : char}
   127→    </span>
   128→  ));
   129→
   130→  const El = tag as any;
   131→  return <El ref={ref} className={`blur-text${revealed ? ' revealed' : ''} ${className}`} style={{ contain: 'layout paint' }}>{chars}</El>;
   132→}

   450→  // Scroll detection — rAF-throttled to avoid layout thrashing.

   454→    const SECTION_IDS = ['home', 'services', 'courses', 'contact'];
   455→    const sectionCache: Record<string, { top: number; height: number }> = {};
   456→
   457→    const measureSections = () => {
   458→      for (const id of SECTION_IDS) {
   459→        const el = document.getElementById(id);
   460→        if (el) {
   461→          sectionCache[id] = { top: el.offsetTop, height: el.offsetHeight };
   462→        }
   463→      }
   464→    };
   465→    measureSections();

   467→    // Re-measure on resize (debounced) — layout shifts invalidate offsets
   468→    let resizeTimer: ReturnType<typeof setTimeout> | null = null;
   469→    const onResize = () => {
   470→      if (resizeTimer) clearTimeout(resizeTimer);
   471→      resizeTimer = setTimeout(measureSections, 200);
   472→    };

   475→    let rafId: number | null = null;

   477→      if (rafId !== null) return;

   479→        rafId = null;

   481→        setScrolled(scrollY > 80);

   484→        const scrollPos = scrollY + 150;
   485→        for (const id of SECTION_IDS) {
   486→          const cached = sectionCache[id];
   487→          if (cached && scrollPos >= cached.top && scrollPos < cached.top + cached.height) {
   488→            setActiveNav(id);
   489→            break;
   490→          }
   491→        }
   492→      });
   493→    };
   494→    window.addEventListener('scroll', onScroll, { passive: true });
   495→    // Re-measure after fonts/images settle
   496→    const settleTimer = setTimeout(measureSections, 600);
   497→    return () => {
   498→      window.removeEventListener('scroll', onScroll);
   499→      window.removeEventListener('resize', onResize);
   500→      if (rafId !== null) cancelAnimationFrame(rafId);
   501→      if (resizeTimer) clearTimeout(resizeTimer);
   502→      clearTimeout(settleTimer);
   503→    };
